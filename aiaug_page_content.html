function getAIAugmentationPage() { // Extract AI security services from the main services page const allServices = getServicesPage(); const aiServices = allServices.split('
')[1].split('
')[0] .split('
') .filter(card => card.includes('AI Security') || card.includes('fa-robot') || card.includes('machine learning') || card.includes('ML') || card.includes('chatbot') || card.includes('anomaly')) .map(card => { // Convert service-card to ai-feature-card format return card.replace('service-card', 'ai-feature-card') .replace('service-icon', '') .replace('service-price', '') .replace('service-category', ''); }) .map(card => '
' + card.split('
')[1]) .join(''); return `

AI Augmentation Services

100+ AI-powered security services to supercharge your defenses

${aiServices}
`; }
