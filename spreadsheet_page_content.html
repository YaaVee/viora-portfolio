function getSpreadsheetPage() { // Extract spreadsheet automation services from the main services page const allServices = getServicesPage(); const spreadsheetServices = allServices.split('
')[1].split('
')[0] .split('
') .filter(card => card.includes('Spreadsheet Automation') || card.includes('file-excel') || card.includes('compliance') || card.includes('tracking') || card.includes('dashboard')) .map(card => { // Convert service-card to automation-card format return card.replace('service-card', 'automation-card') .replace('service-icon', 'automation-icon') .replace('service-category', '') .replace('service-price', '') .replace('service-desc', ''); }) .map(card => '
' + card.split('
')[1]) .join(''); return `

Spreadsheet Automation Services

100+ specialized spreadsheet automation services for security teams

${spreadsheetServices}
`; }
