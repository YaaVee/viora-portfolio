function getCloudSecurityPage() { // Extract cloud security services from the main services page const allServices = getServicesPage(); const cloudServices = allServices.split('
')[1].split('
')[0] .split('
') .filter(card => card.includes('Cloud Security') || card.includes('fab fa-aws') || card.includes('fab fa-microsoft') || card.includes('fab fa-google') || card.includes('fa-docker') || card.includes('Kubernetes') || card.includes('Container')) .map(card => '
' + card) .join(''); return `

Cloud Security Services

100+ specialized cloud security services across all major platforms

${cloudServices}
`; }
